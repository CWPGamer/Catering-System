Function SignIn()
	Repeat
		Print "1. Sign in as student/parent, 2. Sign in as canteen staff"
		Print "3. Return to Main Menu"
		Read choice

		If (choice == 1) Then
			call SignInStudent()
		Else If (choice == 2) Then
			call SignInStaff()
		Else IF (choice == 3)
			return
		Else
			Print "Invalid Input"
		Endif
	Until choice == 3	
EndFunction

SubFunction SignInStudent()
	email, password as string
	i, maxSize, success as integer
	success <-- 0
	maxSize <-- 5
	
	Repeat			
		For i = 1 to (maxSize) Do
			Print "Enter email address: "
			Read email
			Print "Enter password: "
			Read password
			
			If (email == StuEmail[i] AND password == StuPWord[i]) Then
				Print "Sign in successful!!"
				success <-- 1
				i = maxSize + 1		
			Else
				Print "Invalid email"
			Endif
		Endfor
	Until success == 1
	Call Student()
EndFunction
